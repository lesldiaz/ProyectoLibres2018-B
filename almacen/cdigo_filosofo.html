Código Filosofo

package filosofos;

 

public class Filosofo implements Runnable{

    private Object tenedorIzq;

    private Object tenedorDer;

 

    public Filosofo(Object tenedorIzq, Object tenedorDer) {

        this.tenedorIzq = tenedorIzq;

        this.tenedorDer = tenedorDer;

    }

    /*

    Metodo que imprime la accion con el nombre del hilo y lo duerme durante un tiempo random

    */

     private void doAction(String action) throws InterruptedException {

          System.out.println(

          Thread.currentThread().getName() + " " + action);

          Thread.sleep(((int) (Math.random() * 100)));

    }

    

    @Override

    public void run() {

        try {

            while (true) {

                doAction(": Esta pensando");

                synchronized (tenedorIzq){ //si el palillo izq esta desocupado

                    doAction(": Toma el tenedor izquierdo");

                synchronized (tenedorDer){ //si el palillo der esta desocupado

                        doAction(": Toma el tenedor derecho || Esta comiendo");

                        doAction(": Baja el tenedor derecho");

                    }

                    // Regresa a pensar

                    doAction(": Baja el tenedor izquierdo || Regresa a pensar");

                }

            }

        } catch (InterruptedException e) {

            Thread.currentThread().interrupt();

            System.err.print("Ocurrió un error con el filosofo " + Thread.currentThread().getName());

            return;

        }

   

    }  

}
