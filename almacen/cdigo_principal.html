Código Principal

package filosofos;

 

public class Principal {

 

    public static void main(String[] args) {

       

        //Arreglo de 5 filósofos

        final Filosofo[] filocomen = new Filosofo[2];

        //Arreglo de 5 tenedores/palillos, uno por cada filósofo

        Object[] tenedores = new Object[filocomen.length];

        //Declaración de tenedores/palillos dentro del arreglo

        for (int i = 0; i < tenedores.length; i++) {

            tenedores[i] = new Object();

        }

        /*

        Asignacion de tenedores a cada uno de los filosofos

        */

        for (int i = 0; i < filocomen.length; i++) {

            Object tenedorIzq = tenedores[i];

            Object tenedorDer = tenedores[(i + 1) % tenedores.length];

            if (i == filocomen.length - 1) {

                // el ultimo filosofo toma el palillo derecho primero

                filocomen[i] = new Filosofo(tenedorDer, tenedorIzq);

            } else {

                //Asignación normal para los demas filósofos

                filocomen[i] = new Filosofo(tenedorIzq, tenedorDer);

            }

             //Declaracion de hilos para los filósofos

           Thread t = new Thread(filocomen[i], "Filósofo " + (i + 1));

            t.start();

        }

    }

}
